122 void readEC()
123 {
124  float Vin;

126  float raw;

129  raw = analogRead(EC_PIN);
130  raw = analogRead(EC_PIN);

132 
133  Vin = getVin();
134  Vdrop = (Vin * raw) / 1024.0;
135  Rc = (Vdrop * (R1 + PIN_R)) / (Vin - Vdrop);
136  Rc = Rc - PIN_R;
